#include <iostream>
using namespace std;
double addition(double num1, double num2)
{
    double sum = 0;
    sum = num1 + num2;

    return sum;
}

double subtraction(double num1, double num2)
{
    double sub = 0;
    sub = num1 - num2;
    return sub;
}

double multiplication(double num1, double num2)
{

    double mul = 0;
    mul = num1 * num2;
    return mul;
}
double division(double num1, double num2)
{

    double div = 0;

    div = num1 / num2;

    return div;
}

int main()
{
    cout << "Enter operation:\n";

            cin >> x;

        else if (h == 's' || h == 'S')
